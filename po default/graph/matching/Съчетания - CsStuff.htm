Въведение

Задачата за намиране на максимално съчетание в двуделен и недвуделен граф има много практически приложения. Едно от тях е така наречената задача за назначенията. Нека разполагаме с определен брой машини и списък от задачи които трябва да се изпълнят от тези машини. Като всяка машина изпълнява определена задача точно за един ден и може да изпълнява не повече от една задача едновременно. Също така знаем коя машина кои задачи може да изпълни. Да се намери колко задачи най-много могат да бъдат изпълнени за един ден и съответното разписание на изпълнение.

Дефиниране на проблема

Даден е неориентиран граф , съчетание наричаме подмножество от ребра такова че за всеки връх , има най-много едно ребро от коeто е свързвано с . Също така казваме че връх участва в съчетанието ако съществува ребро от на което единият край е , в противен случаи неучаства. Максималено съчетание е такова в което участват най-голям брой ребра, казваме че съчетание е максимално, ако за всяко друго съчетание е изпълнено . Нека приемем че множеството от върхове може да бъде разделено на , където и всяко ребро свързва връх и връх .


Например ние може да разглеждаме съчетание на множество от машини с множество от задачи които трябва да бъдат изпълнение едновременно. Нека приемем, че присъствието на ребро означава че съответната машина е способна да извърши задача . Максималното съчетание позволява едновременната работа на колкото се може повече машини.

Алгоритъм

Даден е двуделен граф , където и за всички ребра е изпълнено че и . Също така нека е някакво съчетание в . Казваме че простия път е нарастващ път с влияние върху ако той започва от връх който не участва в и завършва във връх който също не участва в M, също така ребрата от този път трябва да принадлежат алтернативно на и .

Сега нека разменим ребрата от този път така, че всяко ребро което не е било от вече е от него и обратно.

По този начин получаваме ново съчетание в което ребрата са с едно повече.

Тук ще покажем псевдо код на алгоритъм които намира максималното съчетание в двуделен граф. За даден граф , нека и

Алгоритъма които намира такова съчетание:

1. В началото не сме свързали никои с никои и множеството

2. Чрез обхождане в ширина намираме нарастващ път започващ от текущо разглеждания връх

3. Ако такъв път има то обръщаме ребрата от този път като всяко ребро което е участвало в вече не участва и обратно

4. Повтаряме от стъпка 2 за следващия връх


Така описаният алгоритъм работи не само за двуделен но и за произволен граф. Сложност на алгоритъма е , защото намирането на нарастващ път отнема време пропорционално на и ние търсим точно нарастващи пътя.

Примерна реализация 


/*
Формат на входния файл:
На първия ред са записани 3 числа -  - съответно броя на върховете в ,  и броя на ребрата от L към R.
Следват  реда на които има по 2 числа - съответно връх от  и връх от  които са свързани

Примерен вход("matching.in"):
5 4 8
1 6
2 6
2 8
3 7
3 8
3 9
4 8
5 8

*/


#include <stdio.h>
#include <queue>
#include <vector>
using namespace std;

const int N = 1024;
const int NONE = -2;

int n1, n2, m;

vector<int> a[N];


void Input(){
	freopen("matching.in", "rt", stdin);
	scanf("%d%d%d", &n1, &n2, &m);
	for(int i=0; i<m; i++){
		int p, q;
		scanf("%d%d", &p, &q);
		p--;
		q--;
		a[p].push_back(q);
	}
}

void Matching(){
	
	
	int match[N];
	bool used[N];
	int prev[N];
	queue<int> q;
	int i, j;

	for(i=0; i<n1+n2; i++)
		match[i] = NONE;
	

	for(i=0; i<n1; i++){
		
		for(j=0; j<n1+n2; j++){
			used[j] = false;
		}
		
		q.push(i);
		prev[i] = NONE;

		bool found = false;
		while(!q.empty() && !found){
			int tmp = q.front();
			used[tmp] = 1;
			for(j=0; j<a[tmp].size(); j++){
				int next = a[tmp][j];
				if(!used[ next ]){			
					used[ next ] = true;						
					prev[ next ] = tmp;
					q.push(next);
					if(match[ next ] == NONE){							
						found = true;
						break; 
					}
					else{
						prev[ match[ next ] ] = next;
						q.push( match[ next ] );
					}				
				}
			}
			if(found){
				tmp = q.back();
				while(prev[tmp] != NONE){
					if(tmp >= n1)
						match[tmp] = prev[tmp];
					tmp = prev[tmp];					
				}
			}
			else{
				q.pop();
			}

		}
		while(!q.empty())
			q.pop();
	}
	int max = 0;
	for(i=n1; i<n1+n2; i++)
		if(match[i] != NONE)
			max++;
	printf("Maximum matching is %d.\n", max);
	for(i=n1; i<n1+n2; i++)
		if(match[i] != NONE)
			printf("%d - %d\n", match[i]+1, i+1);
	printf("\n");
}

int main(){
	Input();
	Matching();
	return 0;
}
